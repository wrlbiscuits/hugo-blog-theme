原题链接:[4382. 快速打字 - AcWing题库](https://www.acwing.com/problem/content/4385/)

若字符串b能通过删减得到a,则a是b的子串

双指针遍历,最后检查a是否被完全遍历即可
```c++
#include <cstdio>
#include <string>
#include <iostream>
using namespace std;
int N;
int main(){
	scanf("%d",&N);
	for(int k = 1;k <= N;k++){
		string s1,s2;
		cin >> s1 >> s2;
		int pos = 0;
		int x = s1.size();
		int y = s2.size();
		for(int i = 0;i < y;i++){
			if(s2[i] == s1[pos]) pos++;
			if(pos == x) break;
		}
		if(pos == x) printf("Case #%d: %d\n",k,y - x);
		else printf("Case #%d: IMPOSSIBLE\n",k);
	}
	return 0;
}
```
